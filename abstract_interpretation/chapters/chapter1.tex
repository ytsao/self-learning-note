\begin{theorem}
    For all $x \in \R$, we have $x^2 \geq 0$.
\end{theorem}

\begin{proof}
    This follows from the fact that any real number multiplied by itself is non-negative.
\end{proof}

\begin{example}
    Consider $x = -3$. Then $x^2 = 9 > 0$.
\end{example}
